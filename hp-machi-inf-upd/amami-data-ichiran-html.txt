    <table id="dataTable">
        <thead>
            <tr>
                <th>市町村コード</th>

                <th>名称</th>

        <tbody></tbody>

    <script>
        let dataList = [];

        document.getElementById("dataForm").addEventListener("submit", function(event) {
            event.preventDefault();
            
            const entry = {
                municipality: document.getElementById("municipality").value,
                genre: document.getElementById("genre").value,
                subcategory: document.getElementById("subcategory").value,
                name: document.getElementById("name").value,
                overview: document.getElementById("overview").value,
                zipcode: document.getElementById("zipcode").value,
                address: document.getElementById("address").value,
                phone: document.getElementById("phone").value
            };

            dataList.push(entry);
            displayData();
        });

        function displayData() {
            const tableBody = document.querySelector("#dataTable tbody");
            tableBody.innerHTML = "";
            dataList.forEach((data, index) => {
                let row = `<tr>
                    <td>${data.municipality}</td>
                    <td>${data.genre}</td>
                    <td>${data.subcategory}</td>
                    <td>${data.name}</td>
                    <td>${data.overview}</td>
                    <td>${data.address}</td>
                    <td>${data.phone}</td>
                    <td><button onclick="editData(${index})">編集</button></td>
                    <td><button onclick="deleteData(${index})">削除</button></td>
                </tr>`;
                tableBody.innerHTML += row;
            });
        }

        function deleteData(index) {
            dataList.splice(index, 1);
            displayData();
        }

        function editData(index) {
            const entry = dataList[index];
            document.getElementById("municipality").value = entry.municipality;
            document.getElementById("genre").value = entry.genre;
            document.getElementById("subcategory").value = entry.subcategory;
            document.getElementById("name").value = entry.name;
            document.getElementById("overview").value = entry.overview;
            document.getElementById("zipcode").value = entry.zipcode;
            document.getElementById("address").value = entry.address;
            document.getElementById("phone").value = entry.phone;
            dataList.splice(index, 1);
            displayData();
        }

        function searchData() {
            const keyword = document.getElementById("searchBar").value.toLowerCase();
            const filteredData = dataList.filter(entry => Object.values(entry).some(value => value.toLowerCase().includes(keyword)));
            
            document.querySelector("#dataTable tbody").innerHTML = "";
            filteredData.forEach((data, index) => {
                let row = `<tr>
                    <td>${data.municipality}</td>
                    <td>${data.genre}</td>
                    <td>${data.subcategory}</td>
                    <td>${data.name}</td>
                    <td>${data.overview}</td>
                    <td>${data.address}</td>
                    <td>${data.phone}</td>
                    <td><button onclick="editData(${index})">編集</button></td>
                    <td><button onclick="deleteData(${index})">削除</button></td>
                </tr>`;
                document.querySelector("#dataTable tbody").innerHTML += row;
            });
        }
